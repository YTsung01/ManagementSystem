	//依據empId查詢目前加班的累積時數
	@Override
	public List<OverTime> findOverTimeHourByEmpId(Integer empId) {
		String sql = "SELECT SUM(overTimeHour) AS totalOverTime FROM managementsystem.overTimeList WHERE empId = ? && verifyState = 1";
		List<OverTime> OverTimes = jdbcTemplate.query(sql, new BeanPropertyRowMapper<>(OverTime.class), empId);
		return OverTimes;
	}
